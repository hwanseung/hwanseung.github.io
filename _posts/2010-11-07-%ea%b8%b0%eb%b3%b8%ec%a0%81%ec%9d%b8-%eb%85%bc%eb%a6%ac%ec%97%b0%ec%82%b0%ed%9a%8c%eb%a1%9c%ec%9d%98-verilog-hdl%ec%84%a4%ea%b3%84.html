//** Basic_logic_circuit ************************
module Basic_logic_circuit (Ai,Bi,S1,S0,Ei);
  output Ei;
  input Ai,Bi,S1,S0;

  wire w1,w2,w3,w4;

  and (w1,Ai,Bi);
  or (w2,Ai,Bi);
  xor (w3,Ai,Bi);
  not (w4,Ai);

  Mux_4to1 mux4_1(w1,w2,w3,w4,S1,S0,Ei);

endmodule


//** Mux_4to1( ************************
module Mux_4to1(I0,I1,I2,I3,S1,S0,Y);
  output Y;
  input I0,I1,I2,I3,S1,S0;

  wire w1,w2,w3,w4,w5,w6;
  
  not (w6,S1);
  not (w5,S0);
  and (w1,I0,w6,w5);
  and (w2,I1,w6,S0);
  and (w3,I2,S1,w5);
  and (w4,I3,S1,S0);
  or (Y,w1,w2,w3,w4);
   
endmodule
